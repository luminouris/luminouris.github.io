2022 寒假集训线段树

发表于2022-01-16更新于2022-01-19阅读次数

高中学过，所以思路大概还记得，但是代码实现忘了一些，就一些。

先上复杂度，修改为logn，查询为logn，所以复杂度为O（nlogn），比朴素算法的O（n^2)快了不少。
应用：主要是区间修改和区间询问。

下面是线段树代码的各个部分

一、建树

先开数组mi[]表示各个节点，第n个节点的左孩子节点为2n，右孩子节点为2n+1
，首个节点为1，表示整个区间。

void build(int id,int l,int r)
{
        if(l==r) //回溯条件：当前节点为叶结点
    {
        sum[id]=num[l];
        return;
    }
    int mid=(l+r)>>1;//位运算优化常数
    build(id<<1,l,mid);//左子树构造
    build(id<<1|1,mid+1,r);//右子树构造
    sum[id]=sum[id<<1]+sum[id<<1|1];//更新父节点，各个题目条件不同，此处为区间和
}

二、lazy tag与区间修改

在区间修改时倘若我们采取将区间内的每个点修改再回溯，无异于再重新建造一棵线段树，复杂度为O（mnlogn），甚至高于了朴素算法，因此我们需要引入lazy tag，将复杂度维持在O（nlogn）

void pushdown(int id,int l,int r,int ql,int qr)
{
    if(add[id]==0)
    return;
    int mid=(l+r)>>1;
    add[id<<1]+=add[id];
    sum[id<<1]+=(mid-l+1)*add[id];
    add[id<<1|1]+=add[id];
    sum[id<<1|1]+=(r-mid)*add[id];
    add[id]=0;
}
void update(int id,int l,int r,int ql,int qr,int v)
{
    if(l>qr || r<ql)//回溯条件两区间无交集
    return;
    if(l>=ql && r<=qr)
    {
        add[id]+=v;
        sum[id]+=(r-l+1)*v;
        return;
    }//询问区间包含当前区间，更新lazy tag和sum后即可回溯
    pushdown(id,l,r,ql,qr);
    int mid=(l+r)>>1;
    update(id<<1,l,mid,ql,qr,v);
    update(id<<1|1,mid+1,r,ql,qr,v);
    sum[id]=sum[id<<1]+sum[id<<1|1];
}

三、区间查询

int query(int id,int l,int r,int ql,int qr)
 {
    if(l>qr || r<ql)//无交集返回0；
    return 0;
    if(l>=ql && r<=qr)//完全包含；
    return sum[id];
    pushdown(id,l,r,ql,qr);
    int mid=(l+r)>>1;
    return         query(id<<1,l,mid,ql,qr)+query(id<<1|1,mid+1,r,ql,qr);//部分包含；
 }
